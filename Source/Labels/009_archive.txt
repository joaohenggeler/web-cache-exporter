# This label is a catch-all for several Unix utility compression formats.
# See: http://fileformats.archiveteam.org/wiki/Compress_(Unix)#The_0x1f_compression_family
BEGIN_FILE Compress

	BEGIN_SIGNATURES
		# Compact
		1F FF
		FF 1F

		# Compress
		1F 9D

		# Freeze/Melt (old format)
		1F 9E
		# Freeze/Melt (new format)
		1F 9F

		# Pack
		1F 1E
		# Pack (old format)
		1F 1F

		# Quasijarus Strong Compression
		1F A1

		# SCO Compress LZH
		1F A0
	END

	BEGIN_MIME_TYPES
		application/x-compress
	END

	BEGIN_EXTENSIONS
		# Compress, Pack, Quasijarus Strong Compression, SCO Compress LZH
		Z
	END

END

BEGIN_FILE ZIP

	BEGIN_SIGNATURES
		# "PK.."
		50 4B 03 04
		# Empty archive
		50 4B 05 06
		# Split or spanned archive
		50 4B 07 08
	END

	BEGIN_MIME_TYPES
		application/zip
		application/x-zip-compressed
	END

	BEGIN_EXTENSIONS
		zip
	END

END

BEGIN_FILE Zlib

	BEGIN_SIGNATURES
		# Assumes 0x78 as the first byte (DEFLATE with a 32K window).

		# No compression (level 0)
		78 01
		# Best speed (level 1-5)
		78 5E
		# Default compression (level 6)
		78 9C
		# Best compression (level 7-9)
		78 DA
	END

END

BEGIN_FILE Zstandard

	BEGIN_SIGNATURES

		37 A4 30 EC
	END

	BEGIN_MIME_TYPES
		application/zstd
		application/zstandard
	END

	BEGIN_EXTENSIONS
		zst
	END
